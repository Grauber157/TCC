CREATE DATABASE math_language;
USE math_language;
CREATE TABLE usuario (
	id_aluno INT AUTO_INCREMENT,
    email VARCHAR(255),
    senha VARCHAR(255),
    nome_usuario VARCHAR(50),
    apelido_usuario VARCHAR(50),
    descricao VARCHAR(250),
    data_nascimento DATE,
    pontuacao INT(10),
    ano_escolar VARCHAR(10),
    instituicao_escolar VARCHAR(50),
    turma_codigo VARCHAR(4),
    PRIMARY KEY (id_aluno)
);

CREATE TABLE turma (
	codigo VARCHAR(4),
    nome_turma VARCHAR(50),
    senha_turma VARCHAR(255),
    pontuacao_geral INT(20),
    PRIMARY KEY (codigo)
);

CREATE TABLE jogos (
	id_usuario INT,
	escolha_portas INT,
    ache_o_x INT
);


ALTER TABLE jogos ADD FOREIGN KEY (id_usuario) REFERENCES usuario(id_aluno);

ALTER TABLE usuario ADD FOREIGN KEY (turma_codigo) REFERENCES turma(codigo)
